Using \(\vec r(t)=\langle 4\cos(2t),4\sin(2t),6t\rangle\),
\begin{enumerate}[label=\alph*.]
  \item Find \(\vec T(t)\)
  \item Find \(\vec N(t)\)
  \item Find the curvature
\end{enumerate}
===
\begin{enumerate}[label=\alph*.]
  \item \(\vec T(t)=
    \langle-\frac{4}{5}\sin(2t),\frac{4}{5}\cos(2t),\frac{3}{5}\rangle\)
  \item \(\vec N(t)=
    \langle-\cos(2t),\sin(2t),0\rangle\)
  \item \(k=\frac{4}{25}\)
\end{enumerate}
